CREATE OR REPLACE FUNCTION GET_RECENT_ACCID
RETURN NUMBER
IS ACCID NUMBER;
BEGIN
  SELECT MAX(ACCT_ID) INTO ACCID
  FROM ACCOUNT;
  RETURN ACCID;
END;
